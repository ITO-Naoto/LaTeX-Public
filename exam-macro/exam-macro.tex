%マクロ定義開始

\newcounter{Mode}
\newcommand{\mode}[1]{\setcounter{Mode}{#1}}
\mode{1}

\newcommand{\ans}[1]{%
    \ifnum\theMode=1 
    \textcolor{red}{#1}%
    \fi
    \ifnum\theMode=2 
    \newbox\answ
    \setbox\answ=\hbox{#1}
    \kern\the\wd\answ%
    \fi
    \ifnum\theMode=3
    \fi
}

\newcommand{\kaisetu}[1]{%
    \ifnum\theMode=1 
    \begin{itembox}[l]{解答}
        #1
    \end{itembox}%
    \fi
    \ifnum\theMode=2
    \begin{itembox}[l]{解答}
        #1
    \end{itembox}
    \fi
    \ifnum\theMode=3
    \fi
}
